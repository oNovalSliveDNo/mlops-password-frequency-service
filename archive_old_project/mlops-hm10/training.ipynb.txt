```python
def build_monitoring_tests(timestamp, data_date):
    return TestSuite(
        tests=[
            # 8 обязательных тестов из LMS
            TestValueMAPE(lt=16),
            TestValueRMSE(lt=5.1e5),
            TestShareOfMissingValues(lt=0.0155),

            TestNumberOfConstantColumns(eq=0),
            TestNumberOfEmptyRows(eq=0),
            TestNumberOfEmptyColumns(eq=0),
            TestNumberOfDuplicatedColumns(eq=0),

            TestShareOfDriftedColumns(lt=0.7),

            # 2 дополнительных теста
            TestNumberOfRows(gt=0),
            TestNumberOfColumns(eq=reference_for_tests.shape[1]),
        ],
        timestamp=timestamp,
        tags=["daily_tests", str(data_date)],
        metadata={
            "report_type": "daily_tests",
            "data_date": str(data_date),
            "report_date": str(timestamp.date()),
        },
    )



daily_test_suites = {}

for date, group in current_for_tests.groupby(current_for_tests["start_date"].dt.date):
    report_timestamp = pd.Timestamp(date) + pd.Timedelta(days=1)
    
    test_suite = build_monitoring_tests(
        timestamp=report_timestamp.to_pydatetime(),
        data_date=date,
    )
    
    test_suite.run(
        reference_data=reference_for_tests,
        current_data=group,
        column_mapping=column_mapping,
    )
    
    ws.add_test_suite(project.id, test_suite)
    daily_test_suites[date] = test_suite
```
